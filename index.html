// Objeto global para armazenar quando cada veículo começou a ficar parado let cronometroParada = {}; db.ref('monitoramento').on('value', (snap) => { const data = snap.val(); const lista = document.getElementById('lista-visitantes'); lista.innerHTML = ""; let alertaGeral = false; const AGORA = Date.now(); if (!data) { document.getElementById('cont-v').innerText = "0"; gerenciarAlarme(false); return; } document.getElementById('cont-v').innerText = Object.keys(data).length; Object.keys(data).forEach(id => { const info = data[id]; const vPos = { lat: info.lat, lng: info.lng }; // --- LÓGICA DE FILTRO DE 60 SEGUNDOS --- const velocidadeBaixa = (info.velocidade < 2); if (velocidadeBaixa) { // Se ele acabou de parar, registra o horário do início da parada if (!cronometroParada[id]) { cronometroParada[id] = AGORA; } } else { // Se ele se moveu, limpa o cronômetro dele delete cronometroParada[id]; } // Só considera "Realmente Parado" se o tempo decorrido for > 60.000ms (60s) const tempoParadoMs = cronometroParada[id] ? (AGORA - cronometroParada[id]) : 0; const estaParadoALERTA = velocidadeBaixa && (tempoParadoMs > 60000); if (estaParadoALERTA || info.status === "fora_da_rota") { alertaGeral = true; } // Gerenciar Marcador Google if (!markersV[id]) { markersV[id] = new google.maps.Marker({ position: vPos, map: map, icon: { path: google.maps.SymbolPath.CIRCLE, scale: 8, fillColor: estaParadoALERTA ? "#f59e0b" : "#10b981", fillOpacity: 1, strokeWeight: 2, strokeColor: "white" } }); } else { markersV[id].setPosition(vPos); markersV[id].setOptions({ icon: { path: google.maps.SymbolPath.CIRCLE, scale: 8, fillColor: estaParadoALERTA ? "#f59e0b" : "#10b981", fillOpacity: 1, strokeWeight: 2, strokeColor: "white" } }); } const card = document.createElement('div'); card.className = `card ${estaParadoALERTA ? 'parado-alerta' : ''}`; // Mostra contagem regressiva ou tempo parado no Card para o operador ver let infoTempo = ""; if (velocidadeBaixa && !estaParadoALERTA) { const faltam = Math.ceil((60000 - tempoParadoMs) / 1000); infoTempo = `
Verificando parada: ${faltam}s...`; } else if (estaParadoALERTA) { const total = Math.round(tempoParadoMs / 60000); infoTempo = `
⚠️ PARADO HÁ ${total} MIN`; } card.innerHTML = ` ${estaParadoALERTA ? 'PARADO' : 'EM ROTA'}

${info.visitante}

🚗 PLACA: ${info.placa} | 🚀 VEL: ${info.velocidade} km/h

📍 ${info.lote_destino}${infoTempo}

`; card.onclick = () => map.panTo(vPos); lista.appendChild(card); }); gerenciarAlarme(alertaGeral); });
